Implementation Code Examples

1. Initialize Supabase Client

// supabaseClient.js import { createClient } from '@supabase/supabase-js'; const supabaseUrl = process.env.SUPABASE_URL; const supabaseAnonKey = process.env.SUPABASE_ANON_KEY; export const supabase = createClient(supabaseUrl, supabaseAnonKey, { auth: { autoRefreshToken: true, persistSession: true, detectSessionInUrl: true } });

2. Fetch User Profile

// Fetch complete user data from Supabase async function getUserProfile(userId) { const { data, error } = await supabase .from('profiles') .select(` id, username, email, subscription:subscriptions ( plan, status ) `) .eq('id', userId) .single(); if (error) throw error; return data; }

3. Server-Side Authentication Middleware

// Express middleware for Supabase authentication async function supabaseAuthMiddleware(req, res, next) { const token = req.headers.authorization?.substring(7); if (!token) { return res.status(401).json({ error: 'Missing authorization token' }); } const userData = await verifyAndGetUserData(token); req.supabaseUser = userData; req.userContext = { username: userData.username, plan: userData.plan, ip: req.ip }; next(); }

4. Secure Endpoint Example

// Secure chat endpoint with user context app.post('/api/chat', supabaseAuthMiddleware, async (req, res) => { const { message } = req.body; const userData = req.supabaseUser; // Process message with user context const response = `Hello ${userData.username}! ` + `As a ${userData.plan} user, you have access to: ` + userData.permissions.join(', '); res.json({ success: true, response: response, user: { username: userData.username, plan: userData.plan } }); } );

üîí Security Best Practices

‚úÖ DO:

  • Store credentials in environment variables
  • Use Row Level Security (RLS) policies in Supabase
  • Validate all user input on both client and server
  • Implement rate limiting to prevent abuse
  • Use HTTPS for all API communications
  • Implement token refresh mechanisms
  • Log security events for monitoring
  • Regularly update dependencies

‚ùå DON'T:

Never:
  • Commit API keys or secrets to version control
  • Use service role keys on the client side
  • Trust client-side data without server validation
  • Store sensitive data in local storage
  • Expose internal error details to users
  • Skip input sanitization
